28/// Updates the ebpf "policy_map" with an id and a config. This can
29/// be either an existing id or a new a id.
30pub fn update_kernel_policy_map(
31    maps: &SeaBeeMapHandles,
32    id: u32,
33    config: &policy_file::PolicyConfig,
34) -> Result<()> {
35    let config = config.to_c_policy_config();
36    if let Err(e) =
37        maps.policy_map
38            .update(id.as_bytes(), config.as_bytes(), libbpf_rs::MapFlags::ANY)
39    {
40        return Err(anyhow!(
41            "Error: update_kernel_policy_map: possibly exceeded max policies: {}\n error: '{}'",
42            constants::SEABEE_MAX_POLICIES,
43            e
44        ));
45    }
46    trace!("Completed update_kernel_policy_map for id: {id}\n{config:?}");
47    Ok(())
48}

50/// removes a policy from the kernel based on an id.
51///
52/// This removes the policy from the policy map so there will no longer
53/// be a policy associated with the given policy id. Objects still labeled with this id
54/// will be treated as if they have no policy id.
55pub fn remove_kernel_policy(maps: &SeaBeeMapHandles, policy_id: u32) -> Result<()> {
56    debug!("remove kernel policy: {}", policy_id);
57    maps.policy_map.delete(policy_id.as_bytes())?;
58
59    // inode and task local storage labels will persist,
60    //   but will not be associated with a policy, so they won't be affected
61    // map protections will be removed when maps are deallocated
62    Ok(())
63}

65// Add a path to the scope corresponding to an id
66pub fn add_path_to_scope(maps: &SeaBeeMapHandles, path: &Path, id: u32) -> Result<()> {
67    let key = utils::str_to_bytes(
68        path.to_str()
69            .context(format!("failed to covert {} to string", path.display()))?,
70        PATH_MAX.try_into()?,
71    )?;
72
73    if let Err(e) =
74        maps.path_to_pol_id
75            .update(key.as_slice(), id.as_bytes(), libbpf_rs::MapFlags::ANY)
76    {
77        return Err(anyhow!(
78            "Error: add_path_to_scope: possibly exceeded max policy scopes: {}\n error: '{e}'",
79            constants::SEABEE_MAX_POLICY_SCOPES,
80        ));
81    }
82    debug!("added path {} to scope for policy {}", path.display(), id);
83    Ok(())
84}

86/// Label all files for a policy with a given id
87pub fn label_files_for_policy(policy: &PolicyFile, sb_maps: &SeaBeeMapHandles) -> Result<()> {
88    debug!("label_files_for_policy id: {}", policy.id);
89
90    // ensure all files exist before labeling any of them
91    for path in &policy.files {
92        if !path.exists() {
93            return Err(anyhow!("label_files_for_policy: path '{}' does not exist.\nAll paths must exist on policy load. See the policy documentation.", path.display()));
94        }
95    }
96
97    // label files
98    let mut open_object = MaybeUninit::uninit();
99    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
100    for path in &policy.files {
101        utils::walk_with(path, |entry| {
102            trigger_inode_labeling(&skel.maps.filename_to_policy_id, entry.path(), policy.id)
103                .map_err(|e| {
104                    anyhow!(
105                        "label_files_for_policy {}: failed to label {}\n{e}",
106                        policy.id,
107                        path.display()
108                    )
109                })?;
110
111            Ok(())
112        })?;
113    }
114
115    Ok(())
116}

118/// Label a given file with a given id
119pub fn label_file_with_id(sb_maps: &SeaBeeMapHandles, path: &Path, id: u32) -> Result<()> {
120    debug!("Labeling file {} with id: {id}", path.display());
121    let mut open_object = MaybeUninit::uninit();
122    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
123    if let Err(e) = trigger_inode_labeling(&skel.maps.filename_to_policy_id, path, id) {
124        return Err(anyhow!(
125            "label_file_with_id failed on {}: {e}",
126            path.display()
127        ));
128    }
129
130    Ok(())
131}

133/// remove label on a given file
134pub fn unlabel_file(sb_maps: &SeaBeeMapHandles, file: &PathBuf) -> Result<()> {
135    debug!("unlabel file {}", file.display());
136    let mut open_object = MaybeUninit::uninit();
137    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
138    trigger_inode_labeling(
139        &skel.maps.filename_to_policy_id,
140        Path::new(file),
141        bpf::seabee::NO_POL_ID,
142    )?;
143
144    Ok(())
145}

147// checks the seabee config and labels pins if protect_pins is true
148pub fn label_seabee_pins(
149    pins: &Vec<PinnedLink>,
150    protect_pins: bool,
151    sb_maps: &SeaBeeMapHandles,
152) -> Result<()> {
153    // check if pins should be protected
154    if !protect_pins {
155        return Ok(());
156    }
157
158    debug!("Labeling pins...");
159    let mut open_object = MaybeUninit::uninit();
160    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
161
162    for pin in pins {
163        match pin.link.pin_path() {
164            Some(path) => {
165                trigger_inode_labeling(&skel.maps.filename_to_policy_id, &path, BASE_POLICY_ID)?
166            }
167            None => {
168                return Err(anyhow!("link was not pinned!"));
169            }
170        }
171    }
172    Ok(())
173}

175/// Adds access protections for every map in a skeleton
176pub fn label_maps_for_skel(
177    skel: &dyn Skel,
178    policy_id: u32,
179    sb_maps: &SeaBeeMapHandles,
180) -> Result<()> {
181    debug!("Labeling maps...");
182
183    // Setup
184    let num_maps = skel.object().maps().count();
185    let mut open_object = MaybeUninit::uninit();
186    let label_skel = load_label_maps_skel(&mut open_object, sb_maps, num_maps)?;
187    for map in skel.object().maps() {
188        let key = map.info()?.info.id;
189        label_skel.maps.map_id_to_pol_id.update(
190            key.as_bytes(),
191            policy_id.as_bytes(),
192            libbpf_rs::MapFlags::ANY,
193        )?;
194    }
195
196    // Trigger eBPF via 'bpf_map_get_fd_by_id'
197    // This will trigger for every map on the system, but only the SeaBee maps will be labeled
198    for _map in query::MapInfoIter::default() {
199        continue;
200    }
201
202    Ok(())
203}

205// should only be called after load_label_file_skel(), this triggers
206// loaded eBPF programs which actually execute the file labeling
207fn trigger_inode_labeling(map: &Map, path: &Path, policy_id: u32) -> Result<()> {
208    // Update map with filename
209    let file_name = path.file_name().context(format!(
210        "trigger_file_labeling: file_name() failed on {}",
211        path.display()
212    ))?;
213    let key = crate::utils::str_to_bytes(
214        file_name.to_str().context(format!(
215            "trigger_file_labeling: to_str() failed on {}",
216            path.display()
217        ))?,
218        128,
219    )?;
220    map.update(
221        key.as_slice(),
222        policy_id.as_bytes(),
223        libbpf_rs::MapFlags::ANY,
224    )?;
225
226    // trigger 'seabee_label_file' eBPF program in 'label_file.bpf.c' using stat.
227    // After successfully labeling the file, a permission denied is returned.
228    // We don't actually care about the result of stat.
229    // The denial simply indicates we successfully triggered our program.
230    match nix::sys::stat::stat(path) {
231        Ok(_) => Err(anyhow!(
232            "trigger_file_labeling: stat returned unexpected allow for {}",
233            path.display()
234        )),
235        Err(e) => {
236            if e == nix::errno::Errno::EPERM {
237                Ok(())
238            } else {
239                Err(anyhow!("trigger_file_labeling: failed stat: {e}"))
240            }
241        }
242    }?;
243
244    // Reset map
245    map.delete(key.as_slice())?;
246    Ok(())
247}

249/// Updates task storage for associated pid with the given policy id
250pub fn label_seabee_process(sb: &SeaBee) -> Result<()> {
251    debug!("Labeling seabee process with id: {BASE_POLICY_ID}");
252    let mut open_object = MaybeUninit::uninit();
253    let _skel = load_label_task_skel(&mut open_object, sb, BASE_POLICY_ID)?;
254
255    // signal each thread in task
256    for entry in fs::read_dir("/proc/self/task")? {
257        if let Some(name) = entry?.file_name().to_str() {
258            let tid = name.parse::<u32>().map_err(|e| {
259                anyhow!("label_seabee_process: failed to convert {name} to u32: {e}")
260            })?;
261            let tgid = process::id();
262
263            // signal thread and catch error
264            let result =
265                unsafe { libc::syscall(libc::SYS_tgkill, tgid as i32, tid as i32, libc::SIGCONT) };
266            if result != 0 {
267                return Err(anyhow!(
268                    "libc::kill returned error: {}",
269                    std::io::Error::last_os_error()
270                ));
271            }
272        }
273    }
274
275    Ok(())
276}

278fn load_label_file_skel<'a>(
279    open_object: &'a mut MaybeUninit<OpenObject>,
280    sb_maps: &SeaBeeMapHandles,
281) -> Result<label_file::LabelFileSkel<'a>> {
282    let mut open_skel = label_file::LabelFileSkelBuilder::default().open(open_object)?;
283
284    open_skel
285        .maps
286        .inode_storage
287        .reuse_fd(sb_maps.inode_storage.as_fd())?;
288    open_skel
289        .maps
290        .log_ringbuf
291        .reuse_fd(sb_maps.log_ringbuf.as_fd())?;
292    open_skel
293        .maps
294        .policy_map
295        .reuse_fd(sb_maps.policy_map.as_fd())?;
296    open_skel.maps.bss_data.user_pid = process::id();
297    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
298    let mut skel = open_skel.load()?;
299    skel.attach()?;
300
301    debug!("Label file skel was attached");
302
303    Ok(skel)
304}

306fn load_label_task_skel<'a>(
307    open_object: &'a mut MaybeUninit<OpenObject>,
308    sb: &SeaBee,
309    policy_id: u32,
310) -> Result<label_task::LabelTaskSkel<'a>> {
311    let mut open_skel = label_task::LabelTaskSkelBuilder::default().open(open_object)?;
312    open_skel
313        .maps
314        .task_storage
315        .reuse_fd(sb.maps.task_storage.as_fd())?;
316    open_skel
317        .maps
318        .log_ringbuf
319        .reuse_fd(sb.maps.log_ringbuf.as_fd())?;
320    open_skel
321        .maps
322        .policy_map
323        .reuse_fd(sb.maps.policy_map.as_fd())?;
324    open_skel.maps.bss_data.user_pid = process::id();
325    open_skel.maps.bss_data.policy_id = policy_id;
326    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
327    let mut skel = open_skel.load()?;
328    skel.attach()?;
329
330    debug!("Label task skel was attached");
331
332    Ok(skel)
333}

335fn load_label_maps_skel<'a>(
336    open_object: &'a mut MaybeUninit<OpenObject>,
337    sb_maps: &SeaBeeMapHandles,
338    num_maps: usize,
339) -> Result<label_maps::LabelMapsSkel<'a>> {
340    let mut open_skel = label_maps::LabelMapsSkelBuilder::default().open(open_object)?;
341    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
342    open_skel
343        .maps
344        .map_to_pol_id
345        .reuse_fd(sb_maps.map_to_pol_id.as_fd())?;
346    open_skel
347        .maps
348        .log_ringbuf
349        .reuse_fd(sb_maps.log_ringbuf.as_fd())?;
350    open_skel
351        .maps
352        .map_id_to_pol_id
353        .set_max_entries(num_maps.try_into()?)?;
354    let mut skel = open_skel.load()?;
355    skel.attach()?;
356
357    debug!("Label maps skel was attached");
358
359    Ok(skel)
