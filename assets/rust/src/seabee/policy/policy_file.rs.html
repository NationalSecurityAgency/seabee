20pub trait FromYamlPath: DeserializeOwned {
21    fn from_path(path: &Path) -> Result<Self> {
22        utils::verify_file_has_ext(path, vec!["yaml", "yml"])?;
23        let request_str = match std::fs::read_to_string(path) {
24            Ok(s) => s,
25            Err(e) => return Err(anyhow!("error reading '{}' to string: {e}", path.display())),
26        };
27        Ok(serde_yaml::from_str(&request_str)?)
28    }
29}
30
31impl<T> FromYamlPath for T where T: DeserializeOwned {}

33#[derive(Clone, Debug, Deserialize, Serialize, PartialEq)]
34#[serde(default)]
35#[serde(deny_unknown_fields)]
36#[serde_as]
37pub struct PolicyConfig {
38    /// Select map security level
39    pub map_access: SecurityLevel,
40    /// Wether or not pins should be tracked by SeaBee
41    pub include_pins: bool,
42    /// Security level for files and pins
43    pub file_write_access: SecurityLevel,
44    /// Determines if how ptrace can be used on processes in scope
45    pub ptrace_access: SecurityLevel,
46    /// Determines how to apply signal mask
47    pub signal_access: SecurityLevel,
48    /// Determines which signals should be allowed
49    #[serde_as(as = "DisplayFromStr")] // allows hex formatting
50    pub signal_allow_mask: u64,
51}
52
53/// Use block as default for security
54/// These defaults are used for the SeaBee Config as well
55impl Default for PolicyConfig {
56    fn default() -> Self {
57        Self {
58            map_access: SecurityLevel::block,
59            include_pins: true,
60            file_write_access: SecurityLevel::block,
61            ptrace_access: SecurityLevel::block,
62            signal_access: SecurityLevel::block,
63            // generate a sigmask for all signals that can kill a process
64            signal_allow_mask: utils::generate_sigmask(false),
65        }
66    }
67}
68
69impl PolicyConfig {
70    pub fn to_c_policy_config(&self) -> bpf::seabee::c_policy_config {
71        bpf::seabee::c_policy_config {
72            signal_allow_mask: self.signal_allow_mask,
73            signal_access: self.signal_access as u8,
74            ptrace_access: self.ptrace_access as u8,
75            file_write_access: self.file_write_access as u8,
76            map_access: self.map_access as u8,
77            protect_pins: if self.include_pins { 1_u8 } else { 0_u8 },
78            padding_1: 0,
79            padding_2: 0,
80            padding_3: 0,
81        }
82    }
83}

85#[derive(Clone, Debug, Default, Deserialize, Serialize)]
86#[serde(deny_unknown_fields)]
87pub struct PolicyFile {
88    /// Uniquely identifies a policy file and corresponds to name
89    #[serde(skip, default = "default_policy_id")]
90    pub id: u32,
91    // Stores the path to the policy in the seabee policy directory
92    #[serde(skip, default)]
93    pub seabee_path: PathBuf,
94    /// Uniquely identifies a policy file
95    pub name: String,
96    pub version: u32,
97    pub scope: HashSet<PathBuf>,
98    pub files: HashSet<PathBuf>,
99    pub config: PolicyConfig,
100    #[serde(default)]
101    pub digest: crypto::SeaBeeDigest,
102    // Identifies the key used to sign this policy
103    #[serde(skip)]
104    pub key_id: Option<u32>,
105}
106
107fn default_policy_id() -> u32 {
108    NO_POLICY_ID
109}
110
111impl PolicyFile {
112    pub fn base(config: &Config) -> Result<Self> {
113        let current_exe = std::env::current_exe()?;
114        Ok(Self {
115            id: BASE_POLICY_ID,
116            name: String::from("base"),
117            files: HashSet::from([
118                current_exe,
119                PathBuf::from(constants::SERVICE_PATH),
120                PathBuf::from(constants::SEABEE_DIR),
121                PathBuf::from(constants::SEABEECTL_EXE),
122                PathBuf::from(constants::PIN_DIR),
123                // SOCKET_PATH does not exist on policy load
124            ]),
125            config: config.policy_config.clone(),
126            ..Default::default()
127        })
128    }
129
130    pub fn display_short(&self) -> String {
131        format!(
132            "{}({}) scope: {}",
133            self.name,
134            self.id,
135            self.scope.iter().map(|p| p.display()).join(", ")
136        )
137    }
138}

140impl std::fmt::Display for PolicyFile {
141    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
142        let key_id_str = match self.key_id {
143            Some(id) => id.to_string(),
144            None => String::from("None"),
145        };
146        write!(
147            f,
148            "{}: {}\n  signed by key id: {}\n  version: {}\n  scope: {}\n  files: {}\n  config:\n    maps: {}\n    pins: {}\n    files: {}\n    ptrace: {}\n    signals: {}\n    signal allow mask: {}",
149            self.id, self.name, key_id_str, self.version, self.scope.iter().map(|p| p.display()).join(", "), self.files.iter().map(|p| p.display()).join(", "), self.config.map_access, self.config.include_pins, self.config.file_write_access, self.config.ptrace_access, self.config.signal_access, self.config.signal_allow_mask
150        )
151    }
152}
153
154impl PartialEq for PolicyFile {
155    fn eq(&self, other: &Self) -> bool {
156        self.name == other.name
157    }
158}
